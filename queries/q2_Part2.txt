SELECT ?cname (count (?cname) as ?count)
WHERE {
?sub a focu:Course.
?sub focu:courseName ?cname.
?lecture dcterms:isPartOf ?sub.
?lecture foaf:topic ?topicid.
?topicid rdfs:label "AI".
}
GROUP BY ?cname ORDER BY DESC(?count)
